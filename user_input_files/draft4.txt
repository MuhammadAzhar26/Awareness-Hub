// --- LOGIN SYSTEM ---
function loginUser() {
  const username = document.getElementById('username').value.trim();
  const password = document.getElementById('password').value.trim();
  const feedback = document.getElementById('loginFeedback');

  if (username === "" || password === "") {
    feedback.textContent = "Please enter both username and password.";
    return;
  }

  // Simple local validation (for demo)
  if (password === "cyber123") {
    localStorage.setItem('awarenessUser', username);
    document.getElementById('loginScreen').style.display = 'none';
    document.getElementById('mainContent').style.display = 'block';
    document.getElementById('totalPoints').textContent = 350;
    feedback.textContent = "";
  } else {
    feedback.textContent = "Incorrect password. Try 'cyber123'.";
  }
}

// Auto-login if user already logged in
window.onload = function() {
  const user = localStorage.getItem('awarenessUser');
  if (user) {
    document.getElementById('loginScreen').style.display = 'none';
    document.getElementById('mainContent').style.display = 'block';
    document.getElementById('totalPoints').textContent = 350;
  }
};

// --- OPEN MODULES IN NEW WINDOW ---
function openModule(moduleId) {
  const moduleWindows = {
    cryptography: "modules/cryptography.html",
    passwords: "modules/passwords.html",
    phishing: "modules/phishing.html"
  };

  const url = moduleWindows[moduleId];
  if (url) {
    window.open(url, "_blank");
  } else {
    alert("Module not found or under development.");
  }
}
